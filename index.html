💡 Usage Example

// After loading Highcharts and the plugin
Highcharts.chart('container', {
    title: {
        text: 'Chart Title',
        events: {
            click: function() {
                console.log('Title clicked!');
            },
            dblclick: function() {
                console.log('Title double-clicked!');
            }
        }
    }
});
